安装python3 you-get ossutil



 #安装you-get
 dnf -y install python3
 pip3 install you-get


 #安装 git go 
 dnf -y install git go

 #安装ossutil
 go get github.com/aliyun/ossutil

配置环境变量


cd /etc/profile.d/

vim gopath.sh

##gopath.sh内容
PATH=/root/go/bin:$PATH
export PATH
##gopath.sh内容


source /etc/profile

配置ossutil，设置endpoit、accesskey

ossutil如需要etstoken(直接回车跳过) 

ossutil config
ossutil ls
ossutil cp 1.txt oss://3ew223
