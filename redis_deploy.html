1、Standalone 配置文件


#ip ip 端口
bind 172.17.100.156
port 6379

#密码
requirepass password

#进程id 和日志文件
pidfile /var/run/redis_6379.pid
logfile "/var/log/redis_6379.log"

#最大使用内存和回收策略
maxmemory 400MB
maxmemory-policy allkeys-random
